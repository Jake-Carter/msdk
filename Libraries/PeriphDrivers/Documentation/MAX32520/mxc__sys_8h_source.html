121typedef struct {
122 int ie_status;
123 int in_critical;
124} mxc_crit_state_t;
125
126static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
127
128static inline void _mxc_crit_get_state()
129{
130#ifndef __riscv
131 /*
132 On ARM M the 0th bit of the Priority Mask register indicates
133 whether interrupts are enabled or not.
134
135 0 = enabled
136 1 = disabled
137 */
138 uint32_t primask = __get_PRIMASK();
139 _state.ie_status = (primask == 0);
140#else
141 /*
142 On RISC-V bit position 3 (Machine Interrupt Enable) of the
143 mstatus register indicates whether interrupts are enabled.
144
145 0 = disabled
146 1 = enabled
147 */
148 uint32_t mstatus = get_mstatus();
149 _state.ie_status = ((mstatus & (1 << 3)) != 0);
150#endif
151}
152
156static inline void MXC_SYS_Crit_Enter(void)
157{
158 _mxc_crit_get_state();
159 if (_state.ie_status)
160 __disable_irq();
161 _state.in_critical = 1;
162}
163
168static inline void MXC_SYS_Crit_Exit(void)
169{
170 if (_state.ie_status) {
171 __enable_irq();
172 }
173 _state.in_critical = 0;
174 _mxc_crit_get_state();
175 /*
176 ^ Reset the state again to prevent edge case
177 where interrupts get disabled, then Crit_Exit() gets
178 called, which would inadvertently re-enable interrupts
179 from old state.
180 */
181}
182
188static inline int MXC_SYS_In_Crit_Section(void)
189{
190 return _state.in_critical;
191}
