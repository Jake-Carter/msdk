198typedef struct {
199 int ie_status;
200 int in_critical;
201} mxc_crit_state_t;
202
203static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
204
205static inline void _mxc_crit_get_state()
206{
207#ifndef __riscv
208 /*
209 On ARM M the 0th bit of the Priority Mask register indicates
210 whether interrupts are enabled or not.
211
212 0 = enabled
213 1 = disabled
214 */
215 uint32_t primask = __get_PRIMASK();
216 _state.ie_status = (primask == 0);
217#else
218 /*
219 On RISC-V bit position 3 (Machine Interrupt Enable) of the
220 mstatus register indicates whether interrupts are enabled.
221
222 0 = disabled
223 1 = enabled
224 */
225 uint32_t mstatus = get_mstatus();
226 _state.ie_status = ((mstatus & (1 << 3)) != 0);
227#endif
228}
229
241static inline void MXC_SYS_Crit_Enter(void)
242{
243 _mxc_crit_get_state();
244 if (_state.ie_status)
245 __disable_irq();
246 _state.in_critical = 1;
247}
248
253static inline void MXC_SYS_Crit_Exit(void)
254{
255 if (_state.ie_status) {
256 __enable_irq();
257 }
258 _state.in_critical = 0;
259 _mxc_crit_get_state();
260 /*
261 ^ Reset the state again to prevent edge case
262 where interrupts get disabled, then Crit_Exit() gets
263 called, which would inadvertently re-enable interrupts
264 from old state.
265 */
266}
267
273static inline int MXC_SYS_In_Crit_Section(void)
274{
275 return _state.in_critical;
276}
