191typedef struct {
192 int ie_status;
193 int in_critical;
194} mxc_crit_state_t;
195
196static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
197
198static inline void _mxc_crit_get_state()
199{
200#ifndef __riscv
201 /*
202 On ARM M the 0th bit of the Priority Mask register indicates
203 whether interrupts are enabled or not.
204
205 0 = enabled
206 1 = disabled
207 */
208 uint32_t primask = __get_PRIMASK();
209 _state.ie_status = (primask == 0);
210#else
211 /*
212 On RISC-V bit position 3 (Machine Interrupt Enable) of the
213 mstatus register indicates whether interrupts are enabled.
214
215 0 = disabled
216 1 = enabled
217 */
218 uint32_t mstatus = get_mstatus();
219 _state.ie_status = ((mstatus & (1 << 3)) != 0);
220#endif
221}
222
226static inline void MXC_SYS_Crit_Enter(void)
227{
228 _mxc_crit_get_state();
229 if (_state.ie_status)
230 __disable_irq();
231 _state.in_critical = 1;
232}
233
238static inline void MXC_SYS_Crit_Exit(void)
239{
240 if (_state.ie_status) {
241 __enable_irq();
242 }
243 _state.in_critical = 0;
244 _mxc_crit_get_state();
245 /*
246 ^ Reset the state again to prevent edge case
247 where interrupts get disabled, then Crit_Exit() gets
248 called, which would inadvertently re-enable interrupts
249 from old state.
250 */
251}
252
258static inline int MXC_SYS_In_Crit_Section(void)
259{
260 return _state.in_critical;
261}
