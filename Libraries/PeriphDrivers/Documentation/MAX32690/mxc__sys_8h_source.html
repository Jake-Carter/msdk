213typedef struct {
214 int ie_status;
215 int in_critical;
216} mxc_crit_state_t;
217
218static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
219
220static inline void _mxc_crit_get_state()
221{
222#ifndef __riscv
223 /*
224 On ARM M the 0th bit of the Priority Mask register indicates
225 whether interrupts are enabled or not.
226
227 0 = enabled
228 1 = disabled
229 */
230 uint32_t primask = __get_PRIMASK();
231 _state.ie_status = (primask == 0);
232#else
233 /*
234 On RISC-V bit position 3 (Machine Interrupt Enable) of the
235 mstatus register indicates whether interrupts are enabled.
236
237 0 = disabled
238 1 = enabled
239 */
240 uint32_t mstatus = get_mstatus();
241 _state.ie_status = ((mstatus & (1 << 3)) != 0);
242#endif
243}
244
256static inline void MXC_SYS_Crit_Enter(void)
257{
258 _mxc_crit_get_state();
259 if (_state.ie_status)
260 __disable_irq();
261 _state.in_critical = 1;
262}
263
268static inline void MXC_SYS_Crit_Exit(void)
269{
270 if (_state.ie_status) {
271 __enable_irq();
272 }
273 _state.in_critical = 0;
274 _mxc_crit_get_state();
275 /*
276 ^ Reset the state again to prevent edge case
277 where interrupts get disabled, then Crit_Exit() gets
278 called, which would inadvertently re-enable interrupts
279 from old state.
280 */
281}
282
288static inline int MXC_SYS_In_Crit_Section(void)
289{
290 return _state.in_critical;
291}
