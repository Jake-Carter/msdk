209typedef struct {
210 int ie_status;
211 int in_critical;
212} mxc_crit_state_t;
213
214static mxc_crit_state_t _state = { .ie_status = 0xFFFFFFFF, .in_critical = 0 };
215
216static inline void _mxc_crit_get_state()
217{
218#ifndef __riscv
219 /*
220 On ARM M the 0th bit of the Priority Mask register indicates
221 whether interrupts are enabled or not.
222
223 0 = enabled
224 1 = disabled
225 */
226 uint32_t primask = __get_PRIMASK();
227 _state.ie_status = (primask == 0);
228#else
229 /*
230 On RISC-V bit position 3 (Machine Interrupt Enable) of the
231 mstatus register indicates whether interrupts are enabled.
232
233 0 = disabled
234 1 = enabled
235 */
236 uint32_t mstatus = get_mstatus();
237 _state.ie_status = ((mstatus & (1 << 3)) != 0);
238#endif
239}
240
244static inline void MXC_SYS_Crit_Enter(void)
245{
246 _mxc_crit_get_state();
247 if (_state.ie_status)
248 __disable_irq();
249 _state.in_critical = 1;
250}
251
256static inline void MXC_SYS_Crit_Exit(void)
257{
258 if (_state.ie_status) {
259 __enable_irq();
260 }
261 _state.in_critical = 0;
262 _mxc_crit_get_state();
263 /*
264 ^ Reset the state again to prevent edge case
265 where interrupts get disabled, then Crit_Exit() gets
266 called, which would inadvertently re-enable interrupts
267 from old state.
268 */
269}
270
276static inline int MXC_SYS_In_Crit_Section(void)
277{
278 return _state.in_critical;
279}
