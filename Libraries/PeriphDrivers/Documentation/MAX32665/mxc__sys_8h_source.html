221typedef struct {
222 int ie_status;
223 int in_critical;
224} mxc_crit_state_t;
225
226static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
227
228static inline void _mxc_crit_get_state()
229{
230#ifndef __riscv
231 /*
232 On ARM M the 0th bit of the Priority Mask register indicates
233 whether interrupts are enabled or not.
234
235 0 = enabled
236 1 = disabled
237 */
238 uint32_t primask = __get_PRIMASK();
239 _state.ie_status = (primask == 0);
240#else
241 /*
242 On RISC-V bit position 3 (Machine Interrupt Enable) of the
243 mstatus register indicates whether interrupts are enabled.
244
245 0 = disabled
246 1 = enabled
247 */
248 uint32_t mstatus = get_mstatus();
249 _state.ie_status = ((mstatus & (1 << 3)) != 0);
250#endif
251}
252
264static inline void MXC_SYS_Crit_Enter(void)
265{
266 _mxc_crit_get_state();
267 if (_state.ie_status)
268 __disable_irq();
269 _state.in_critical = 1;
270}
271
276static inline void MXC_SYS_Crit_Exit(void)
277{
278 if (_state.ie_status) {
279 __enable_irq();
280 }
281 _state.in_critical = 0;
282 _mxc_crit_get_state();
283 /*
284 ^ Reset the state again to prevent edge case
285 where interrupts get disabled, then Crit_Exit() gets
286 called, which would inadvertently re-enable interrupts
287 from old state.
288 */
289}
290
296static inline int MXC_SYS_In_Crit_Section(void)
297{
298 return _state.in_critical;
299}
