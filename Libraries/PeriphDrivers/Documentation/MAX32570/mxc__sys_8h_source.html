243typedef struct {
244 int ie_status;
245 int in_critical;
246} mxc_crit_state_t;

248static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
249
250static inline void _mxc_crit_get_state()
251{
252#ifndef __riscv
253 /*
254 On ARM M the 0th bit of the Priority Mask register indicates
255 whether interrupts are enabled or not.
256
257 0 = enabled
258 1 = disabled
259 */
260 uint32_t primask = __get_PRIMASK();
261 _state.ie_status = (primask == 0);
262#else
263 /*
264 On RISC-V bit position 3 (Machine Interrupt Enable) of the
265 mstatus register indicates whether interrupts are enabled.
266
267 0 = disabled
268 1 = enabled
269 */
270 uint32_t mstatus = get_mstatus();
271 _state.ie_status = ((mstatus & (1 << 3)) != 0);
272#endif
273}
274
286static inline void MXC_SYS_Crit_Enter(void)
287{
288 _mxc_crit_get_state();
289 if (_state.ie_status)
290 __disable_irq();
291 _state.in_critical = 1;
292}
293
298static inline void MXC_SYS_Crit_Exit(void)
299{
300 if (_state.ie_status) {
301 __enable_irq();
302 }
303 _state.in_critical = 0;
304 _mxc_crit_get_state();
305 /*
306 ^ Reset the state again to prevent edge case
307 where interrupts get disabled, then Crit_Exit() gets
308 called, which would inadvertently re-enable interrupts
309 from old state.
310 */
311}
312
318static inline int MXC_SYS_In_Crit_Section(void)
319{
320 return _state.in_critical;
321}
