152typedef struct {
153 int ie_status;
154 int in_critical;
155} mxc_crit_state_t;
156
157static mxc_crit_state_t _state = { .ie_status = (int)0xFFFFFFFF, .in_critical = 0 };
158
159static inline void _mxc_crit_get_state()
160{
161#ifndef __riscv
162 /*
163 On ARM M the 0th bit of the Priority Mask register indicates
164 whether interrupts are enabled or not.
165
166 0 = enabled
167 1 = disabled
168 */
169 uint32_t primask = __get_PRIMASK();
170 _state.ie_status = (primask == 0);
171#else
172 /*
173 On RISC-V bit position 3 (Machine Interrupt Enable) of the
174 mstatus register indicates whether interrupts are enabled.
175
176 0 = disabled
177 1 = enabled
178 */
179 uint32_t mstatus = get_mstatus();
180 _state.ie_status = ((mstatus & (1 << 3)) != 0);
181#endif
182}
183
195static inline void MXC_SYS_Crit_Enter(void)
196{
197 _mxc_crit_get_state();
198 if (_state.ie_status)
199 __disable_irq();
200 _state.in_critical = 1;
201}
202
207static inline void MXC_SYS_Crit_Exit(void)
208{
209 if (_state.ie_status) {
210 __enable_irq();
211 }
212 _state.in_critical = 0;
213 _mxc_crit_get_state();
214 /*
215 ^ Reset the state again to prevent edge case
216 where interrupts get disabled, then Crit_Exit() gets
217 called, which would inadvertently re-enable interrupts
218 from old state.
219 */
220}
221
227static inline int MXC_SYS_In_Crit_Section(void)
228{
229 return _state.in_critical;
230}
